class Customer {
    // Declare instance variables (accessible by child classes)
    int cost = 40;
    String items = "Tomatoes";

    void purchaseList() {
        System.out.println("Purchased item: " + items);
        System.out.println("Cost: " + cost);
    }
}

public class Mart extends Customer {
    // Overriding values
    int cost = 30;
    String items = "Onions";

    void billing() {
        System.out.println("Billing item: " + items);
        System.out.println("Cost: " + cost);
    }

    public static void main(String[] args) {
        Mart mart = new Mart();

        // Accessing child class fields
        System.out.println("Items: " + mart.items);  // Onions
        System.out.println("Cost: " + mart.cost);    // 30

        System.out.println("%%%%%%%%%%%%%%");

        // Accessing parent class fields via super
        System.out.println("Super items: " + ((Customer)mart).items);  // Tomatoes
        System.out.println("Super cost: " + ((Customer)mart).cost);    // 40

        // Or access using super inside Mart methods
        mart.purchaseList();  // From Customer
        mart.billing();       // From Mart
    }
}
