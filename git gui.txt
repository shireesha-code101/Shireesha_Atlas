import java.io.FileInputStream;
import java.io.IOException;
import java.io.FileNotFoundException;

public class ReadingByte {
    public static void main(String[] args) {
        String fileName = "FileName01.txt";
        int byteRead;

        try (FileInputStream infile = new FileInputStream(fileName)) {
            System.out.println("Reading content from '" + fileName + "':");
            while ((byteRead = infile.read()) != -1) {

                System.out.print((char) byteRead);
            }
            System.out.println("\nEnd of file reached.");
        } catch (FileNotFoundException e) {

            System.err.println("Error: The file '" + fileName + "' was not found.");

        } catch (IOException e) {

            System.err.println("An I/O error occurred while reading from '" + fileName + "': " + e.getMessage());
            e.printStackTrace();
        }
        System.out.println("Thank You...!!!");
    }
}
