
public class ExceptionDemo {

    // Method that explicitly throws IllegalAccessException (checked)
    static void fun() throws IllegalAccessException {
        System.out.println("Inside fun().");
        throw new IllegalAccessException("demo exception by fun method");
    }

    // Example method that may cause ArrayIndexOutOfBoundsException
    static void arr2() throws ArrayIndexOutOfBoundsException {
        int[] arr = {1, 2, 3};
        System.out.println(arr[5]);  // triggers unchecked exception
    }

    // Example method that may cause FileNotFoundException
    // static void method3() throws FileNotFoundException {
    //     FileReader fr = new FileReader("nonexistent.txt");
    // }

    public static void main(String[] args) {
        try {
            fun();
            // arr2();
            // method3();
        } catch (IllegalAccessException e) {
            System.out.println("Caught in main: " + e.getMessage());
        }
        // Unchecked exceptions like ArrayIndexOutOfBoundsException
        // would bubble up unless caught here.
        System.out.println("Main method resuming execution.");
    }
}
