public class NameArray {
    public static void main(String[] args) {
        // Initialize the character array with your name
        char[] name = {'P', 'r', 'a', 'd', 'y', 'u', 't'};

        // Print the character array
        System.out.println(name);

        // Calculate the length of the name
        int n = name.length;

        // Display the number of letters in the name
        System.out.println("There are " + n + " letters in my name.");

        // Traverse the array and print each character
        System.out.print("Each letter in my name: ");
        for (int i = 0; i < n; i++) {
            System.out.print(name[i] + " ");
        }
    }
}
